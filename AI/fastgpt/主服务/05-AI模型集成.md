# FastGPT AI 模型集成

> 本文档详细分析 AI 模型集成模块的前端到后端完整实现

---

## 1. 功能概述

AI 模型集成是 FastGPT 的核心能力层，支持：

- **5 种模型类型**：LLM、Embedding、TTS、STT、Rerank
- **多提供商支持**：OpenAI、Claude、Gemini 等
- **流式处理**：实时文本生成、推理内容解析
- **Agent 循环**：自动工具调用和多轮推理
- **统一计费**：Token 计数和积分消费

```typescript
export async function loadSystemModels(
  init: boolean,
  language: string
): Promise<void> {
  // 1. 从数据库加载模型配置
  const dbModels = await MongoSystemModel.find({ isActive: true }).lean();

  // 2. 从插件加载模型
  const pluginModels = await pluginClient.model.list();

  // 3. 合并模型列表
  const allModels = [...dbModels, ...pluginModels];

  // 4. 按类型分类
  const llmModels: LLMModelItemType[] = [];
  const embeddingModels: EmbeddingModelItemType[] = [];
  const ttsModels: TTSModelType[] = [];
  const sttModels: STTModelType[] = [];
  const rerankModels: RerankModelItemType[] = [];

  for (const model of allModels) {
    switch (model.type) {
      case ModelTypeEnum.llm:
        llmModels.push(formatLLMModel(model));
        break;
      case ModelTypeEnum.embedding:
        embeddingModels.push(formatEmbeddingModel(model));
        break;
      case ModelTypeEnum.tts:
        ttsModels.push(formatTTSModel(model));
        break;
      case ModelTypeEnum.stt:
        sttModels.push(formatSTTModel(model));
        break;
      case ModelTypeEnum.rerank:
        rerankModels.push(formatRerankModel(model));
        break;
    }
  }

  // 5. 构建全局映射
  global.llmModelMap = new Map(
    llmModels.map((m) => [m.model, m])
  );
  global.embeddingModelMap = new Map(
    embeddingModels.map((m) => [m.model, m])
  );
  global.ttsModelMap = new Map(ttsModels.map((m) => [m.model, m]));
  global.sttModelMap = new Map(sttModels.map((m) => [m.model, m]));
  global.reRankModelMap = new Map(rerankModels.map((m) => [m.model, m]));

  // 6. 设置默认模型
  global.systemDefaultModel = {
    llm: llmModels.find((m) => m.isDefault) || llmModels[0],
    datasetTextLLM: llmModels.find((m) => m.datasetProcess) || llmModels[0],

    embedding: embeddingModels.find((m) => m.isDefault) || embeddingModels[0],
    tts: ttsModels[0],
    stt: sttModels[0],
    rerank: rerankModels[0]
  };
}
```

```typescript
export type CreateLLMResponseResult = {
  text: string;
  reasoningText?: string;
  toolCalls?: ToolCallType[];
  usage?: {
    prompt_tokens: number;
    completion_tokens: number;
    total_tokens: number;
  };
  finishReason?: string;
  runningTime: number;
};

export async function createLLMResponse(
  props: CreateLLMResponseProps
): Promise<CreateLLMResponseResult> {
  const {
    model: modelId,
    messages,
    temperature,
    max_tokens,
    stream = false,
    tools,
    tool_choice,
    onStreaming,
    onReasoning,
    onToolCall,
    onToolParam
  } = props;

  const startTime = Date.now();

  // 1. 获取模型配置
  const model = getLLMModel(modelId);

  // 2. 构建请求配置
  const requestConfig = {
    model: model.model,
    messages,
    temperature: temperature ?? model.maxTemperature * 0.4,
    max_tokens: max_tokens ?? model.maxResponse,
    stream,
    ...(tools?.length && { tools, tool_choice }),
    ...model.defaultConfig
  };

  // 3. 获取 API 客户端
  const client = getOpenAIClient(model);

  // 4. 发送请求
  if (stream) {
    return createStreamResponse({
      client,
      requestConfig,
      model,
      onStreaming,
      onReasoning,
      onToolCall,
      onToolParam,
      startTime
    });
  } else {
    return createNonStreamResponse({
      client,
      requestConfig,
      startTime
    });
  }
}
```

### 4.2 流式响应处理

**文件位置**: `packages/service/core/ai/llm/request.ts`

```typescript
async function createStreamResponse({
  client,
  requestConfig,
  model,
  onStreaming,
  onReasoning,
  onToolCall,
  onToolParam,
  startTime
}: StreamResponseParams): Promise<CreateLLMResponseResult> {
  let text = '';
  let reasoningText = '';
  const toolCalls: ToolCallType[] = [];
  let usage: CreateLLMResponseResult['usage'];
  let finishReason: string | undefined;

  // 推理标签状态
  let inReasoningTag = false;
  const reasoningStartTag = '<think>';
  const reasoningEndTag = '</think>';

  const stream = await client.chat.completions.create({
    ...requestConfig,
    stream: true,
    stream_options: { include_usage: true }
  });

  for await (const chunk of stream) {
    const delta = chunk.choices?.[0]?.delta;
    const content = delta?.content || '';

    // 处理推理标签
    if (model.reasoning) {
      const { normalText, reasoningContent, isInTag } = parseReasoningTags(
        content,
        inReasoningTag,
        reasoningStartTag,
        reasoningEndTag
      );

      inReasoningTag = isInTag;

      if (reasoningContent) {
        reasoningText += reasoningContent;
        onReasoning?.(reasoningContent);
      }

      if (normalText) {
        text += normalText;
        onStreaming?.(normalText);
      }
    } else {
      text += content;
      onStreaming?.(content);
    }

    // 处理工具调用
    if (delta?.tool_calls) {
      for (const toolCall of delta.tool_calls) {
        const index = toolCall.index;

        if (!toolCalls[index]) {
          toolCalls[index] = {
            id: toolCall.id || '',
            name: toolCall.function?.name || '',
            params: ''
          };
          onToolCall?.(toolCalls[index]);
        }

        if (toolCall.function?.arguments) {
          toolCalls[index].params += toolCall.function.arguments;
          onToolParam?.(toolCall.function.arguments);
        }
      }
    }

    // 获取完成原因
    if (chunk.choices?.[0]?.finish_reason) {
      finishReason = chunk.choices[0].finish_reason;
    }

    // 获取使用量
    if (chunk.usage) {
      usage = chunk.usage;
    }
  }

  // 解析工具参数
  for (const tool of toolCalls) {
    try {
      tool.parsedParams = JSON.parse(tool.params);
    } catch {
      // 保持原始字符串
    }
  }

  return {
    text,
    reasoningText: reasoningText || undefined,
    toolCalls: toolCalls.length > 0 ? toolCalls : undefined,
    usage,
    finishReason,
    runningTime: Date.now() - startTime
  };
}

/**
 * 解析推理标签
 */
function parseReasoningTags(
  content: string,
  isInTag: boolean,
  startTag: string,
  endTag: string
): {
  normalText: string;
  reasoningContent: string;
  isInTag: boolean;
} {
  let normalText = '';
  let reasoningContent = '';
  let currentInTag = isInTag;

  let remaining = content;

  while (remaining.length > 0) {
    if (currentInTag) {
      // 在推理标签内，查找结束标签
      const endIndex = remaining.indexOf(endTag);
      if (endIndex >= 0) {
        reasoningContent += remaining.slice(0, endIndex);
        remaining = remaining.slice(endIndex + endTag.length);
        currentInTag = false;
      } else {
        reasoningContent += remaining;
        remaining = '';
      }
    } else {
      // 不在推理标签内，查找开始标签
      const startIndex = remaining.indexOf(startTag);
      if (startIndex >= 0) {
        normalText += remaining.slice(0, startIndex);
        remaining = remaining.slice(startIndex + startTag.length);
        currentInTag = true;
      } else {
        normalText += remaining;
        remaining = '';
      }
    }
  }

  return { normalText, reasoningContent, isInTag: currentInTag };
}
```

### 5.1 Agent 循环

**文件位置**: `packages/service/core/ai/llm/agentCall/index.ts`

```typescript
export type RunAgentCallProps = {
  messages: ChatCompletionMessageParam[];
  tools: ChatCompletionTool[];
  model: string;
  maxRunAgentTimes?: number;

  // 回调
  onToolCall?: (tool: ToolCallType) => void;
  onToolResponse?: (response: any) => void;
  onStreaming?: (text: string) => void;
};

export async function runAgentCall({
  messages,
  tools,
  model,
  maxRunAgentTimes = 10,
  onToolCall,
  onToolResponse,
  onStreaming
}: RunAgentCallProps): Promise<{
  text: string;
  toolResponses: ToolResponseType[];
  usage: UsageType;
}> {
  let currentMessages = [...messages];
  const allToolResponses: ToolResponseType[] = [];
  let totalUsage = { prompt_tokens: 0, completion_tokens: 0, total_tokens: 0 };
  let runTimes = 0;

  while (runTimes < maxRunAgentTimes) {
    runTimes++;

    // 1. 调用 LLM
    const response = await createLLMResponse({
      model,
      messages: currentMessages,
      tools,
      tool_choice: 'auto',
      stream: true,
      onStreaming,
      onToolCall
    });

    // 累计 Token
    if (response.usage) {
      totalUsage.prompt_tokens += response.usage.prompt_tokens;
      totalUsage.completion_tokens += response.usage.completion_tokens;
      totalUsage.total_tokens += response.usage.total_tokens;
    }

    // 2. 检查是否有工具调用
    if (!response.toolCalls || response.toolCalls.length === 0) {
      // 没有工具调用，返回结果
      return {
        text: response.text,
        toolResponses: allToolResponses,
        usage: totalUsage
      };
    }

    // 3. 添加助手消息
    currentMessages.push({
      role: 'assistant',
      content: response.text || null,
      tool_calls: response.toolCalls.map((tc) => ({
        id: tc.id,
        type: 'function' as const,
        function: {
          name: tc.name,
          arguments: tc.params
        }
      }))
    });

    // 4. 执行工具调用
    for (const toolCall of response.toolCalls) {
      try {
        const toolResponse = await executeToolCall(toolCall);
        onToolResponse?.(toolResponse);

        allToolResponses.push({
          toolId: toolCall.id,
          name: toolCall.name,
          response: toolResponse
        });

        // 添加工具响应消息
        currentMessages.push({
          role: 'tool',
          tool_call_id: toolCall.id,
          content: JSON.stringify(toolResponse)
        });
      } catch (error) {
        // 工具执行错误
        currentMessages.push({
          role: 'tool',
          tool_call_id: toolCall.id,
          content: JSON.stringify({ error: error.message })
        });
      }
    }
  }

  // 超过最大次数
  throw new Error('Agent exceeded maximum run times');
}
```

**文件位置**: `packages/service/core/ai/embedding/index.ts`

```typescript
export type GetVectorsParams = {
  model: EmbeddingModelItemType;
  input: string | string[];
  type?: 'query' | 'db';
};

export type GetVectorsResult = {
  vectors: number[][];
  tokens: number;
};

export async function getVectorsByText({
  model,
  input,
  type = 'db'
}: GetVectorsParams): Promise<GetVectorsResult> {
  // 1. 规范化输入
  const inputs = Array.isArray(input) ? input : [input];

  if (inputs.length === 0) {
    return { vectors: [], tokens: 0 };
  }

  // 2. 获取客户端
  const client = getOpenAIClient(model);

  // 3. 批量处理
  const batchSize = model.batchSize || 20;
  const allVectors: number[][] = [];
  let totalTokens = 0;

  for (let i = 0; i < inputs.length; i += batchSize) {
    const batch = inputs.slice(i, i + batchSize);

    // 获取配置
    const config = type === 'query' ? model.queryConfig : model.dbConfig;

    // 调用 API
    const response = await client.embeddings.create({
      model: model.model,
      input: batch,
      ...config,
      ...model.defaultConfig
    });

    // 收集结果
    for (const item of response.data) {
      let vector = item.embedding;

      // 归一化处理
      if (model.normalization) {
        vector = normalizeVector(vector);
      }

      allVectors.push(vector);
    }

    totalTokens += response.usage?.total_tokens || 0;
  }

  return {
    vectors: allVectors,
    tokens: totalTokens
  };
}

/**
 * 向量归一化
 */
function normalizeVector(vector: number[]): number[] {
  const norm = Math.sqrt(vector.reduce((sum, v) => sum + v * v, 0));
  if (norm === 0) return vector;
  return vector.map((v) => v / norm);
}
```

```typescript
export async function text2Speech({
  input,
  model,
  voice,
  speed = 1,
  res
}: Text2SpeechParams): Promise<void> {
  const client = getOpenAIClient({ model } as any);

  const response = await client.audio.speech.create({
    model,
    voice,
    input,
    speed,
    response_format: 'mp3'
  });

  // 设置响应头
  res.setHeader('Content-Type', 'audio/mpeg');

  // 流式写入
  const reader = response.body?.getReader();
  if (!reader) throw new Error('No reader');

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    res.write(value);
  }

  res.end();
}
```

```typescript
export type ReRankParams = {
  model: RerankModelItemType;
  query: string;
  documents: Array<{ id: string; text: string }>;
};

export type ReRankResult = {
  results: Array<{
    id: string;
    score: number;
  }>;
  tokens: number;
};

export async function reRankRecall({
  model,
  query,
  documents
}: ReRankParams): Promise<ReRankResult> {
  if (documents.length === 0) {
    return { results: [], tokens: 0 };
  }

  // 构建请求
  const requestUrl = model.requestUrl || `${process.env.OPENAI_BASE_URL}/rerank`;
  const apiKey = model.requestAuth || process.env.CHAT_API_KEY;

  const response = await axios.post(
    requestUrl,
    {
      model: model.model,
      query,
      documents: documents.map((d) => d.text),
      top_n: documents.length
    },
    {
      headers: {
        Authorization: `Bearer ${apiKey}`,
        'Content-Type': 'application/json'
      }
    }
  );

  // 解析结果
  const results = response.data.results.map((r: any) => ({
    id: documents[r.index].id,
    score: r.relevance_score
  }));

  return {
    results: results.sort((a, b) => b.score - a.score),
    tokens: response.data.usage?.total_tokens || 0
  };
}
```

```typescript
export async function dispatchChatCompletion(
  props: DispatchNodeProps
): Promise<DispatchNodeResultType> {
  const {
    params,
    histories,
    workflowStreamResponse,
    stream
  } = props;

  // 1. 获取参数
  const modelName = params[NodeInputKeyEnum.aiModel];
  const systemPrompt = params[NodeInputKeyEnum.aiSystemPrompt];
  const temperature = params[NodeInputKeyEnum.aiChatTemperature];
  const maxToken = params[NodeInputKeyEnum.aiChatMaxToken];
  const userInput = params[NodeInputKeyEnum.userChatInput];
  const quoteText = params[NodeInputKeyEnum.stringQuoteText];
  const visionMode = params[NodeInputKeyEnum.aiChatVision];

  // 2. 获取模型
  const model = getLLMModel(modelName);

  // 3. 构建消息
  const messages: ChatCompletionMessageParam[] = [];

  // 系统提示词
  if (systemPrompt) {
    messages.push({ role: 'system', content: systemPrompt });
  }

  // 知识库引用
  if (quoteText) {
    messages.push({
      role: 'system',
      content: `以下是参考资料，请基于这些资料回答问题：\n\n${quoteText}`
    });
  }

  // 历史消息
  const historyMessages = formatHistoryForLLM(histories, model);
  messages.push(...historyMessages);

  // 用户输入
  if (visionMode && model.vision) {
    // 视觉模式
    const content: ChatCompletionContentPart[] = [
      { type: 'text', text: userInput }
    ];

    // 添加图片
    const imageUrls = params[NodeInputKeyEnum.fileUrlList] || [];
    for (const url of imageUrls) {
      content.push({
        type: 'image_url',
        image_url: { url }
      });
    }

    messages.push({ role: 'user', content });
  } else {
    messages.push({ role: 'user', content: userInput });
  }

  // 4. 调用 LLM
  let answerText = '';
  let reasoningText = '';

  const response = await createLLMResponse({
    model: model.model,
    messages,
    temperature,
    max_tokens: maxToken,
    stream,
    onStreaming: (text) => {
      answerText += text;
      workflowStreamResponse?.({
        event: SseResponseEventEnum.answer,
        data: textAdaptGptResponse({ text })
      });
    },
    onReasoning: (text) => {
      reasoningText += text;
    }
  });

  answerText = response.text || answerText;

  // 5. 返回结果
  return {
    [DispatchNodeResponseKeyEnum.answerText]: answerText,
    [DispatchNodeResponseKeyEnum.reasoningText]: reasoningText || undefined,
    [DispatchNodeResponseKeyEnum.nodeResponse]: {
      tokens: response.usage?.total_tokens,
      model: model.name,
      temperature,
      maxToken
    },
    [DispatchNodeResponseKeyEnum.nodeDispatchUsages]: [{
      tokens: response.usage?.total_tokens || 0,
      moduleName: props.node.name,
      model: model.model
    }],
    data: {
      [NodeOutputKeyEnum.answerText]: answerText
    }
  };
}
```

## 10. 消息压缩

**文件位置**: `packages/service/core/ai/llm/compress/index.ts`

```typescript
export async function compressMessages({
  messages,
  maxTokens,
  model
}: CompressMessagesParams): Promise<ChatCompletionMessageParam[]> {
  // 计算当前 Token 数
  const currentTokens = countMessagesTokens(messages);

  if (currentTokens <= maxTokens) {
    return messages;
  }

  // 需要压缩
  const systemMessages = messages.filter((m) => m.role === 'system');
  const chatMessages = messages.filter((m) => m.role !== 'system');

  // 保留最后几轮对话
  const recentMessages = chatMessages.slice(-4);
  const toCompressMessages = chatMessages.slice(0, -4);

  if (toCompressMessages.length === 0) {
    return messages;
  }

  // 使用 LLM 压缩
  const compressPrompt = `
请将以下对话历史压缩为简洁的摘要，保留关键信息：

${toCompressMessages.map((m) => `${m.role}: ${m.content}`).join('\n')}

请用 JSON 格式返回：{"summary": "压缩后的摘要"}
`;

  const response = await createLLMResponse({
    model: model.model,
    messages: [{ role: 'user', content: compressPrompt }],
    temperature: 0.1,
    max_tokens: 500
  });

  try {
    const { summary } = JSON.parse(response.text);

    return [
      ...systemMessages,
      { role: 'system', content: `历史对话摘要：${summary}` },
      ...recentMessages
    ];
  } catch {
    // 解析失败，直接截断
    return [...systemMessages, ...recentMessages];
  }
}
```
