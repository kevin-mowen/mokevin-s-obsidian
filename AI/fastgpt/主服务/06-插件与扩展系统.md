# FastGPT 插件与扩展系统

## 1. 概述

FastGPT 的插件与扩展系统提供了强大的工具调用能力，支持系统工具、团队工具、MCP工具和HTTP工具等多种形式。这些工具可以在工作流中作为节点使用，极大地扩展了AI Agent的能力边界。

## 2. 架构设计

### 2.1 工具类型层次

```
工具系统
├── 系统工具 (System Tools)
│   ├── 单一工具 (Tool)
│   └── 工具集 (ToolSet)
├── 团队工具 (Team Tools)
│   ├── 插件应用 (Plugin App)
│   └── 工作流应用 (Workflow App)
├── MCP 工具 (Model Context Protocol)
│   ├── MCP 工具集
│   └── MCP 单一工具
└── HTTP 工具 (HTTP Tool)
    ├── OpenAPI 导入
    └── 自定义 HTTP 请求
```

## 5. 工具控制器

### 5.1 获取系统工具列表

```typescript
// packages/service/core/app/tool/controller.ts

// 获取缓存的系统工具
export const getSystemTools = () => getCachedData(SystemCacheKeyEnum.systemTool);

// 获取带安装状态的系统工具
export const getSystemToolsWithInstalled = async ({
  teamId,
  isRoot
}: {
  teamId: string;
  isRoot: boolean;
}) => {
  const [tools, { installedSet, uninstalledSet }] = await Promise.all([
    getSystemTools(),
    MongoTeamInstalledPlugin.find({ teamId, pluginType: 'tool' }, 'pluginId installed')
      .lean()
      .then((res) => {
        const installedSet = new Set<string>();
        const uninstalledSet = new Set<string>();
        res.forEach((item) => {
          if (item.installed) {
            installedSet.add(item.pluginId);
          } else {
            uninstalledSet.add(item.pluginId);
          }
        });
        return { installedSet, uninstalledSet };
      })
  ]);

  return tools.map((tool) => {
    const installed = (() => {
      // 明确安装的
      if (installedSet.has(tool.id)) return true;
      // root用户且未明确卸载的
      if (isRoot && !uninstalledSet.has(tool.id)) return true;
      // 默认安装且未明确卸载的
      if (tool.defaultInstalled && !uninstalledSet.has(tool.id)) return true;
      return false;
    })();

    return { ...tool, installed };
  });
};
```

## 6. 插件执行引擎

### 6.1 插件运行调度

```typescript
// packages/service/core/workflow/dispatch/plugin/run.ts
export const dispatchRunPlugin = async (props: RunPluginProps): Promise<RunPluginResponse> => {
  const {
    node: { pluginId, version },
    runningAppInfo,
    query,
    params: { system_forbid_stream = false, ...data }
  } = props;

  // 1. 解析工具ID
  const { source, pluginId: formatPluginId } = splitCombineToolId(pluginId);

  // 2. 系统工具直接调用工具运行器
  if (source === AppToolSourceEnum.systemTool) {
    return await dispatchRunTool({
      ...props,
      node: {
        ...props.node,
        toolConfig: {
          systemTool: { toolId: formatPluginId }
        }
      }
    });
  }

  // 3. 验证插件权限
  const pluginData = await authPluginByTmbId({
    appId: pluginId,
    tmbId: runningAppInfo.tmbId,
    per: ReadPermissionVal
  });

  // 4. 获取插件运行时数据
  const plugin = await getChildAppRuntimeById({ id: pluginId, versionId: version });

  // 5. 处理插件输入
  const runtimeNodes = storeNodes2RuntimeNodes(
    plugin.nodes,
    getWorkflowEntryNodeIds(plugin.nodes)
  ).map((node) => {
    if (node.flowNodeType === FlowNodeTypeEnum.pluginInput) {
      return {
        ...node,
        inputs: node.inputs.map((input) => {
          let val = data[input.key] ?? input.value;

          // 解密密码类型
          if (input.renderTypeList.includes(FlowNodeInputTypeEnum.password)) {
            val = anyValueDecrypt(val);
          }

          // 处理动态变量 {{currentTime}}
          val = processDynamicValue(val, input);

          return { ...input, value: val };
        })
      };
    }
    return node;
  });

  // 6. 运行工作流
  const { flowResponses, flowUsages, assistantResponses } = await runWorkflow({
    ...props,
    runningAppInfo: {
      id: String(plugin.id),
      name: plugin.name,
      teamId: plugin.teamId || runningAppInfo.teamId,
      tmbId: plugin.tmbId || runningAppInfo.tmbId,
      isChildApp: true
    },
    runtimeNodes,
    runtimeEdges: storeEdges2RuntimeEdges(plugin.edges)
  });

  // 7. 计算使用量
  const usagePoints = await computedAppToolUsage({
    plugin,
    childrenUsage: flowUsages
  });

  // 8. 返回结果
  return {
    data: output?.pluginOutput ?? {},
    assistantResponses: system_forbid_stream ? [] : assistantResponses,
    // ...
  };
};
```

```typescript
// packages/service/core/app/tool/controller.ts
export const refreshSystemTools = async (): Promise<AppToolTemplateItemType[]> => {
  // 1. 从外部API获取工具列表
  const tools = await APIGetSystemToolList();

  // 2. 从数据库加载插件配置

  const dbPlugins = systemToolsArray
```

```typescript
// packages/global/core/app/tool/utils.ts
export const splitCombineToolId = (id: string): {
  source: AppToolSourceEnum;
  pluginId: string;
} => {
  // 格式: source-pluginId 或 纯pluginId
  const match = id.match(/^(system|mcp|http)-(.+)$/);

  if (match) {
    return {
      source: match[1] as AppToolSourceEnum,
      pluginId: match[2]
    };
  }

  // 默认为个人工具
  return {
    source: AppToolSourceEnum.personal,
    pluginId: id
  };
};

export const combineCombineToolId = (
  source: AppToolSourceEnum,
  pluginId: string
): string => {
  if (source === AppToolSourceEnum.personal) {
    return pluginId;
  }
  return `${source}-${pluginId}`;
};
```

支持导入 OpenAPI 规范定义的 HTTP 工具：

```typescript
// HTTP 工具集应用配置
{
  type: AppTypeEnum.httpToolSet,
  toolConfig: {
    httpToolSet: {
      openAPISchema: { /* OpenAPI 3.0 规范 */ },
      baseUrl: 'https://api.example.com',
      headers: {}
    }
  }
}
```
