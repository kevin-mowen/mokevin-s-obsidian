# FastGPT 工作流引擎

> 本文档详细分析工作流引擎的前端到后端完整实现

---

## 1. 功能概述

工作流引擎是 FastGPT 的核心，支持：

- **30+ 节点类型**：覆盖 AI 对话、数据处理、条件控制、工具调用等
- **可视化编排**：基于 ReactFlow 的拖拽式编辑器
- **DAG 执行**：有向无环图拓扑排序执行
- **流式输出**：SSE 实时响应
- **交互式组件**：用户选择、表单输入等暂停节点

```typescript
import ReactFlow, {
  ReactFlowProvider,
  useNodesState,
  useEdgesState,
  addEdge,
  Connection
} from 'reactflow';

export default function WorkflowEditor({
  nodes: initialNodes,
  edges: initialEdges,
  onSave
}: WorkflowEditorProps) {
  const [nodes, setNodes, onNodesChange] = useNodesState(initialNodes);
  const [edges, setEdges, onEdgesChange] = useEdgesState(initialEdges);

  // 节点类型映射
  const nodeTypes = useMemo(() => ({
    [FlowNodeTypeEnum.workflowStart]: WorkflowStartNode,
    [FlowNodeTypeEnum.chatNode]: ChatNode,
    [FlowNodeTypeEnum.datasetSearchNode]: DatasetSearchNode,
    [FlowNodeTypeEnum.ifElseNode]: IfElseNode,
    [FlowNodeTypeEnum.httpRequest468]: HttpRequestNode,
    [FlowNodeTypeEnum.code]: CodeNode,
    [FlowNodeTypeEnum.answerNode]: AnswerNode,
    // ... 其他节点类型
  }), []);

  // 连接处理
  const onConnect = useCallback((connection: Connection) => {
    // 验证连接有效性
    if (!validateConnection(connection, nodes)) {
      return;
    }
    setEdges((eds) => addEdge(connection, eds));
  }, [nodes, setEdges]);

  // 拖放添加节点
  const onDrop = useCallback((event: React.DragEvent) => {
    event.preventDefault();
    const nodeType = event.dataTransfer.getData('nodeType');
    const position = reactFlowInstance.screenToFlowPosition({
      x: event.clientX,
      y: event.clientY
    });

    const newNode = createNode(nodeType, position);
    setNodes((nds) => [...nds, newNode]);
  }, [reactFlowInstance, setNodes]);

  return (
    <ReactFlowProvider>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        nodeTypes={nodeTypes}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onConnect={onConnect}
        onDrop={onDrop}
        onDragOver={(e) => e.preventDefault()}
      >
        <Background />
        <Controls />
        <MiniMap />
      </ReactFlow>
    </ReactFlowProvider>
  );
}
```

### 5.2 核心执行循环

**文件位置**: `packages/service/core/workflow/dispatch/index.ts`

```typescript
async function runWorkflow(props: RunWorkflowProps): Promise<WorkflowResult> {
  const {
    runtimeNodes,
    runtimeEdges,
    variables,
    workflowStreamResponse
  } = props;

  // 存储所有节点的输出
  const nodeOutputs: Record<string, Record<string, any>> = {};

  // 存储所有响应
  const flowResponses: ChatHistoryItemResType[] = [];
  const flowUsages: ChatNodeUsageType[] = [];
  const assistantResponses: AIChatItemValueItemType[] = [];

  // 执行计数（防止无限循环）
  let runTimes = 0;
  const maxRunTimes = props.maxRunTimes || 500;

  // 主执行循环
  while (runTimes < maxRunTimes) {
    runTimes++;

    // 1. 获取可执行节点（所有前置节点已完成）
    const readyNodes = getReadyNodes(runtimeNodes, runtimeEdges, nodeOutputs);

    if (readyNodes.length === 0) {
      // 没有可执行节点，检查是否完成
      if (isWorkflowComplete(runtimeEdges)) {
        break;
      }
      throw new Error('Workflow deadlock detected');
    }

    // 2. 并行执行所有就绪节点
    const nodeResults = await Promise.all(
      readyNodes.map((node) => executeNode(node, props, nodeOutputs))
    );

    // 3. 处理节点结果
    for (const result of nodeResults) {
      const { nodeId, outputs, response, usages, skipHandleId } = result;

      // 存储输出
      nodeOutputs[nodeId] = outputs;

      // 收集响应
      if (response) {
        flowResponses.push(response);
      }
      if (usages) {
        flowUsages.push(...usages);
      }

      // 处理跳过的边
      if (skipHandleId) {
        updateEdgeStatus(runtimeEdges, nodeId, skipHandleId, 'skipped');
      }

      // 更新边状态为已激活
      updateEdgeStatus(runtimeEdges, nodeId, null, 'active');

      // 流式发送节点状态
      workflowStreamResponse?.({
        event: SseResponseEventEnum.flowNodeStatus,
        data: { nodeId, status: 'success' }
      });
    }
  }

  return {
    flowResponses,
    flowUsages,
    assistantResponses,
    newVariables: variables
  };
}
```

```typescript
async function executeNode(
  node: RuntimeNodeItemType,
  props: RunWorkflowProps,
  nodeOutputs: Record<string, Record<string, any>>
): Promise<NodeExecuteResult> {
  const { flowNodeType, nodeId, inputs } = node;

  // 1. 获取处理函数
  const handler = nodeHandlerMap[flowNodeType];
  if (!handler) {
    throw new Error(`Unknown node type: ${flowNodeType}`);
  }

  // 2. 解析输入值（处理引用）
  const resolvedInputs = resolveNodeInputs(inputs, nodeOutputs, props.variables);

  // 3. 执行节点
  try {
    const startTime = Date.now();

    const result = await handler({
      ...props,
      node,
      params: resolvedInputs
    });

    const runningTime = Date.now() - startTime;

    // 4. 构建响应
    return {
      nodeId,
      outputs: result.data || {},
      response: result[DispatchNodeResponseKeyEnum.nodeResponse],
      usages: result[DispatchNodeResponseKeyEnum.nodeDispatchUsages],
      skipHandleId: result[DispatchNodeResponseKeyEnum.skipHandleId],
      runningTime
    };
  } catch (error) {
    // 5. 错误处理
    if (node.catchError) {
      return {
        nodeId,
        outputs: { [NodeOutputKeyEnum.errorText]: error.message },
        error: true
      };
    }
    throw error;
  }
}
```

```typescript
export async function dispatchChatCompletion(
  props: DispatchNodeProps
): Promise<DispatchNodeResultType> {
  const {
    node,
    params,
    histories,
    workflowStreamResponse,
    stream
  } = props;

  // 1. 获取输入参数
  const {
    [NodeInputKeyEnum.aiModel]: modelName,
    [NodeInputKeyEnum.aiSystemPrompt]: systemPrompt,
    [NodeInputKeyEnum.aiChatTemperature]: temperature,
    [NodeInputKeyEnum.aiChatMaxToken]: maxToken,
    [NodeInputKeyEnum.history]: historyCount,
    [NodeInputKeyEnum.userChatInput]: userInput,
    [NodeInputKeyEnum.stringQuoteText]: quoteText
  } = params;

  // 2. 获取模型配置
  const model = getLLMModel(modelName);

  // 3. 构建消息
  const messages: ChatCompletionMessageParam[] = [];

  // 系统提示词
  if (systemPrompt) {
    messages.push({
      role: 'system',
      content: systemPrompt
    });
  }

  // 知识库引用
  if (quoteText) {
    messages.push({
      role: 'system',
      content: `参考资料：\n${quoteText}`
    });
  }

  // 历史消息
  const historyMessages = formatHistoryMessages(histories, historyCount);
  messages.push(...historyMessages);

  // 用户输入
  messages.push({
    role: 'user',
    content: userInput
  });

  // 4. 调用 LLM
  let answerText = '';
  let reasoningText = '';
  let totalTokens = 0;

  const response = await createLLMResponse({
    model: model.model,
    messages,
    temperature: temperature || model.maxTemperature * 0.4,
    max_tokens: maxToken || model.maxResponse,
    stream,
    // 流式回调
    onStreaming: (text) => {
      answerText += text;
      workflowStreamResponse?.({
        event: SseResponseEventEnum.answer,
        data: textAdaptGptResponse({ text })
      });
    },
    onReasoning: (text) => {
      reasoningText += text;
    }
  });

  answerText = response.text || answerText;
  totalTokens = response.usage?.total_tokens || 0;

  // 5. 返回结果
  return {
```

```typescript
    searchMode,
    usingReRank
  });
```

```typescript
export async function dispatchIfElse(
  props: DispatchNodeProps
): Promise<DispatchNodeResultType> {
  const { params, node } = props;

  // 1. 获取条件列表
  const ifElseList: IfElseConditionType[] = params[NodeInputKeyEnum.ifElseList] || [];

  // 2. 评估每个条件
  let matchedIndex = -1;
  for (let i = 0; i < ifElseList.length; i++) {
    const condition = ifElseList[i];
    const result = evaluateCondition(condition, params);

    if (result) {
      matchedIndex = i;
      break;
    }
  }

  // 3. 确定输出路径
  const outputs = node.outputs;
  const skipHandleIds: string[] = [];

  outputs.forEach((output, index) => {
    // 最后一个是 else 分支
    const isElseBranch = index === outputs.length - 1;
    const shouldSkip = matchedIndex === -1
      ? !isElseBranch  // 没匹配则只走 else
      : index !== matchedIndex;  // 匹配了则只走匹配的

    if (shouldSkip) {
      skipHandleIds.push(output.key);
    }
  });

  return {
    [DispatchNodeResponseKeyEnum.skipHandleId]: skipHandleIds,
    [DispatchNodeResponseKeyEnum.nodeResponse]: {
      ifElseResult: matchedIndex >= 0 ? `条件 ${matchedIndex + 1}` : 'else'
    },
    data: {}
  };
}

// 条件评估
function evaluateCondition(
  condition: IfElseConditionType,
  params: Record<string, any>
): boolean {
  const { variable, condition: op, value } = condition;
  const actualValue = params[variable];

  switch (op) {
    case 'equals':
      return actualValue === value;
    case 'notEquals':
      return actualValue !== value;
    case 'contains':
      return String(actualValue).includes(value);
    case 'notContains':
      return !String(actualValue).includes(value);
    case 'greaterThan':
      return Number(actualValue) > Number(value);
    case 'lessThan':
      return Number(actualValue) < Number(value);
    case 'isEmpty':
      return !actualValue || actualValue === '';
    case 'isNotEmpty':
      return !!actualValue && actualValue !== '';
    default:
      return false;
  }
}
```

```typescript
export async function dispatchHttp468Request(
  props: DispatchNodeProps
): Promise<DispatchNodeResultType> {
  const { params } = props;

  // 1. 获取请求配置
  const {
    [NodeInputKeyEnum.httpReqUrl]: url,
    [NodeInputKeyEnum.httpMethod]: method,
    [NodeInputKeyEnum.httpHeaders]: headers,
    [NodeInputKeyEnum.httpParams]: queryParams,
    [NodeInputKeyEnum.httpJsonBody]: jsonBody,
    [NodeInputKeyEnum.httpTimeout]: timeout = 60000
  } = params;

  // 2. 解析 URL 和参数
  const parsedUrl = new URL(url);
  if (queryParams) {
    Object.entries(queryParams).forEach(([key, value]) => {
      parsedUrl.searchParams.append(key, String(value));
    });
  }

  // 3. 构建请求头
  const requestHeaders: Record<string, string> = {
    'Content-Type': 'application/json',
    ...headers
  };

  // 4. 安全检查（禁止内部地址）
  if (isInternalAddress(parsedUrl.hostname)) {
    throw new Error('Access to internal addresses is not allowed');
  }

  // 5. 发送请求
  try {
    const response = await axios({
      method,
      url: parsedUrl.toString(),
      headers: requestHeaders,
      data: method !== 'GET' ? jsonBody : undefined,
      timeout,
      maxContentLength: 10 * 1024 * 1024 // 10MB 限制
    });

    // 6. 解析响应
    const responseData = response.data;

    return {
      [DispatchNodeResponseKeyEnum.nodeResponse]: {
        httpResult: {
          status: response.status,
          headers: response.headers,
          data: responseData
        }
      },
      data: {
        [NodeOutputKeyEnum.httpRawResponse]: responseData,
        [NodeOutputKeyEnum.success]: true
      }
    };
  } catch (error) {
    if (props.node.catchError) {
      return {
        data: {
          [NodeOutputKeyEnum.httpRawResponse]: null,
          [NodeOutputKeyEnum.success]: false,
          [NodeOutputKeyEnum.errorText]: error.message
        }
      };
    }
    throw error;
  }
}
```

```typescript
export const getWorkflowResponseWrite = ({
  res,
  detail,
  streamResponse,
  showNodeStatus
}: {
  res?: NextApiResponse;
  detail?: boolean;
  streamResponse?: boolean;
  showNodeStatus?: boolean;
}) => {
  return ({ event, data }: { event: SseResponseEventEnum; data: any }) => {
    // 检查响应是否可用
    if (!res || res.closed || !streamResponse) return;

    // 非详情模式只发送答案事件
    const answerEvents = { answer: 1, fastAnswer: 1 };
    if (!detail && !answerEvents[event]) return;

    // 检查是否发送节点状态
    const statusEvents = { flowNodeStatus: 1, toolCall: 1, toolParams: 1, toolResponse: 1 };
    if (!showNodeStatus && statusEvents[event]) return;

    // 写入 SSE 响应
    responseWrite({
      res,
      event: detail ? event : undefined,
      data: JSON.stringify(data)
    });
  };
};

// SSE 写入函数
export function responseWrite({
  res,
  event,
  data
}: {
  res: NextApiResponse;
  event?: string;
  data: string;
}) {
  if (event) {
    res.write(`event: ${event}\n`);
  }
  res.write(`data: ${data}\n\n`);
}
```

### 7.3 前端 SSE 处理

**文件位置**: `packages/web/common/fetch/index.ts`

```typescript
export async function streamFetch({
  url,
  data,
  onMessage,
  onFinish,
  onError
}: StreamFetchProps) {
  const response = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data)
  });

  const reader = response.body?.getReader();
  if (!reader) throw new Error('No reader');

  const decoder = new TextDecoder();
  let buffer = '';

  try {
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      buffer += decoder.decode(value, { stream: true });

      // 解析 SSE 事件
      const lines = buffer.split('\n');
      buffer = lines.pop() || '';

      let currentEvent = '';
      for (const line of lines) {
        if (line.startsWith('event: ')) {
          currentEvent = line.slice(7);
        } else if (line.startsWith('data: ')) {
          const data = line.slice(6);
          try {
            const parsed = JSON.parse(data);
            onMessage({
              event: currentEvent || 'message',
              data: parsed
            });
          } catch {
            // 忽略解析错误
          }
        }
      }
    }
    onFinish?.();
  } catch (error) {
    onError?.(error);
  }
}
```
