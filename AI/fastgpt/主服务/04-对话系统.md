# FastGPT 对话系统

> 本文档详细分析对话系统的前端到后端完整实现

---

## 1. 功能概述

对话系统是 FastGPT 的用户交互核心，支持：

- **多轮对话**：上下文管理、历史记录持久化
- **多渠道接入**：网页、API、分享链接、飞书、企业微信等
- **流式响应**：SSE 实时输出
- **文件处理**：图片、文档上传与处理
- **反馈系统**：用户点赞/点踩、管理员标注

### 3.2 对话状态管理

**文件位置**: `projects/app/src/web/core/chat/context/useChatStore.ts`

```typescript
import { create } from 'zustand';
import { streamFetch } from '@fastgpt/web/common/fetch';

type ChatState = {
  messages: ChatItemType[];
  isLoading: boolean;
  abortController: AbortController | null;

  // Actions
  loadChatHistory: (appId: string, chatId: string) => Promise<void>;
  sendMessage: (content: string, files?: File[]) => Promise<void>;
  stopGeneration: () => void;
};

export const useChatStore = create<ChatState>((set, get) => ({
  messages: [],
  isLoading: false,
  abortController: null,

  loadChatHistory: async (appId, chatId) => {
    const { histories } = await getChatHistoryItems({
      appId,
      chatId,
      offset: 0,
      limit: 50
    });
    set({ messages: histories });
  },

  sendMessage: async (content, files) => {
    const { messages } = get();

    // 1. 添加用户消息
    const userMessage: UserChatItemType = {
      obj: ChatRoleEnum.Human,
      value: [{ type: 'text', text: { content } }]
    };

    // 处理文件
    if (files?.length) {
      const fileUrls = await uploadChatFiles(files);
      userMessage.value.push(
        ...fileUrls.map((url, i) => ({
          type: 'file' as const,
          file: { type: 'image', name: files[i].name, url }
        }))
      );
    }

    set({ messages: [...messages, userMessage], isLoading: true });

    // 2. 创建 AbortController
    const abortController = new AbortController();
    set({ abortController });

    // 3. AI 响应消息
    const aiMessage: AIChatItemType = {
      obj: ChatRoleEnum.AI,
      value: []
    };

    // 4. 发送请求（流式）
    await streamFetch({
      url: '/api/v1/chat/completions',
      data: {
        messages: [...messages, userMessage],
        stream: true
      },
      abortSignal: abortController.signal,
      onMessage: ({ event, data }) => {
        if (event === 'answer') {
          // 流式文本
          const textValue = aiMessage.value.find(
            (v) => v.type === 'text'
          );
          if (textValue && textValue.type === 'text') {
            textValue.text.content += data.delta?.content || '';
          } else {
            aiMessage.value.push({
              type: 'text',
              text: { content: data.delta?.content || '' }
            });
          }
          set({ messages: [...messages, userMessage, aiMessage] });
        } else if (event === 'toolCall') {
          // 工具调用
          aiMessage.value.push({
            type: 'tool',
            tool: data
          });
        }
      },
      onFinish: () => {
        set({ isLoading: false, abortController: null });
      },
      onError: (error) => {
        console.error('Chat error:', error);
        set({ isLoading: false, abortController: null });
      }
    });
  },

  stopGeneration: () => {
    const { abortController } = get();
    abortController?.abort();
    set({ isLoading: false, abortController: null });
  }
}));
```

```typescript
export default function ChatInput({
  onSend,
  appId,
  chatConfig
}: ChatInputProps) {
  const [content, setContent] = useState('');
  const [files, setFiles] = useState<File[]>([]);
  const inputRef = useRef<HTMLTextAreaElement>(null);

  // 文件上传配置
  const fileSelectConfig = chatConfig?.fileSelectConfig;
  const canUpload = fileSelectConfig?.canSelectFile || fileSelectConfig?.canSelectImg;

  const handleSend = () => {
    if (!content.trim() && files.length === 0) return;
    onSend(content, files);
    setContent('');
    setFiles([]);
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  return (
    <Box className="chat-input">
      {/* 文件预览 */}
      {files.length > 0 && (
        <HStack mb={2}>
          {files.map((file, index) => (
            <FilePreview
              key={index}
              file={file}
              onRemove={() => setFiles(files.filter((_, i) => i !== index))}
            />
          ))}
        </HStack>
      )}

      <HStack>
        {/* 文件上传按钮 */}
        {canUpload && (
          <FileUploadButton
            accept={getAcceptTypes(fileSelectConfig)}
            onChange={(newFiles) => setFiles([...files, ...newFiles])}
          />
        )}

        {/* 文本输入 */}
        <Textarea
          ref={inputRef}
          value={content}
          onChange={(e) => setContent(e.target.value)}
          onKeyDown={handleKeyDown}
          placeholder="输入消息..."
          rows={1}
          resize="none"
        />

        {/* 发送按钮 */}
        <IconButton
          icon={<SendIcon />}
          onClick={handleSend}
          isDisabled={!content.trim() && files.length === 0}
        />
      </HStack>
    </Box>
  );
}
```

### 4.1 聊天完成 API (v1)

**文件位置**: `projects/app/src/pages/api/v1/chat/completions.ts`

```typescript
import { dispatchWorkFlow } from '@fastgpt/service/core/workflow/dispatch';
import { saveChat } from '@fastgpt/service/core/chat/saveChat';

async function handler(
  req: ApiRequestProps<ChatCompletionBody>,
  res: ApiResponseType
) {
  const {
    messages,
    stream = false,
    chatId,
    appId,
    variables = {}
  } = req.body;

  // ========== 1. 权限认证 ==========
  const authResult = await (async () => {
    // 分享链接认证
    if (req.body.shareId) {
      return authShareChat({
        shareId: req.body.shareId,
        outLinkUid: req.body.outLinkUid,
        chatId
      });
    }

    // API Key / Token 认证
    return authHeaderRequest({
      req,
      appId,
      chatId
    });
  })();

  const { teamId, tmbId, app, authType, responseDetail } = authResult;

  // ========== 2. 消息转换 ==========
  const chatMessages = GPTMessages2Chats(messages);
  const userQuery = chatMessages[chatMessages.length - 1];

  // ========== 3. 加载历史记录 ==========
  const maxHistoryCount = getMaxHistoryLimitFromNodes(app.modules);
  const { histories } = await getChatItems({
    appId: app._id,
    chatId,
    limit: maxHistoryCount
  });

  const newHistories = concatHistories(histories, chatMessages);

  // ========== 4. 生成响应 ID ==========
  const responseChatItemId = new ObjectId().toString();
  const saveChatId = chatId || new ObjectId().toString();

  // ========== 5. 执行工作流 ==========
  const startTime = Date.now();

  const {
    flowResponses,
    flowUsages,
    assistantResponses,
    newVariables,
    system_memories
  } = await dispatchWorkFlow({
    res,
    mode: 'chat',
    runningAppInfo: {
      id: app._id,
      teamId,
      tmbId,
      name: app.name
    },
    runtimeNodes: storeNodes2RuntimeNodes(app.modules),
    runtimeEdges: app.edges,
    variables: {
      ...variables,
      ...chatDetail?.variables
    },
    query: userQuery.value,
    histories: newHistories,
    stream,
    chatId: saveChatId,
    responseChatItemId
  });

  const durationSeconds = (Date.now() - startTime) / 1000;

  // ========== 6. 保存聊天记录 ==========
  await saveChat({
    teamId,
    tmbId,
    appId: app._id,
    chatId: saveChatId,
    source: getSource(authType),
    userQuery,
    assistantResponses,
    flowResponses,
    variables: newVariables,
    responseChatItemId,
    durationSeconds,
    memories: system_memories
  });

  // ========== 7. 返回响应 ==========
  if (!stream) {
    const responseText = assistantResponses
      .map((r) => r.text?.content)
      .filter(Boolean)
      .join('\n');

    return res.json({
      id: saveChatId,
      choices: [{
        message: { role: 'assistant', content: responseText },
        finish_reason: 'stop'
      }]
    });
  }

  // 流式模式已在 dispatchWorkFlow 中处理
}

export default NextAPI(handler);
```

```typescript
async function handler(
  req: ApiRequestProps<GetHistoriesQuery>,
  res: ApiResponseType<ChatHistoryItemType[]>
): Promise<ChatHistoryItemType[]> {
  const { appId, offset = 0, limit = 20 } = req.query;

  const { tmbId } = await authUserPer({ req, authToken: true });

    appId,
    tmbId

    .sort({ top: -1, updateTime: -1 })

    .limit(limit)

    chatId: h.chatId,
    title: h.customTitle || h.title,
    top: h.top,
    updateTime: h.updateTime,
    source: h.source
  }));
```

```typescript
export async function saveChat({
  teamId,
  tmbId,
  appId,
  chatId,
  source,
  userQuery,
  assistantResponses,
  flowResponses,
  variables,
  responseChatItemId,
  durationSeconds,
  memories
}: SaveChatParams): Promise<void> {
  await mongoSessionRun(async (session) => {
    // 1. 获取或创建聊天会话
    let chat = await MongoChat.findOne({ appId, chatId });

    if (!chat) {
      chat = await MongoChat.create([{
        chatId,
        teamId,
        tmbId,
        appId,
        title: userQuery.value[0]?.text?.content?.slice(0, 50) || '新对话',
        source,
        variables
      }], { session });
    } else {
      // 更新会话
      await MongoChat.updateOne(
        { _id: chat._id },
        {
          $set: {
            updateTime: new Date(),
            variables
          }
        },
        { session }
      );
    }

    // 2. 保存用户消息
    const userDataId = new ObjectId().toString();
    await MongoChatItem.create([{
      dataId: userDataId,
      chatId,
      appId,
      obj: ChatRoleEnum.Human,
      value: userQuery.value
    }], { session });

    // 3. 保存 AI 响应
    await MongoChatItem.create([{
      dataId: responseChatItemId,
      chatId,
      appId,
      obj: ChatRoleEnum.AI,
      value: assistantResponses,
      memories,
      durationSeconds,
      citeCollectionIds: extractCiteIds(assistantResponses)
    }], { session });

    // 4. 保存节点响应详情
    if (flowResponses?.length) {
      await MongoChatItemResponse.create([{
        chatItemDataId: responseChatItemId,
        appId,
        chatId,
        data: flowResponses
      }], { session });
    }

    // 5. 移除文件 TTL
    const fileKeys = extractFileKeys(userQuery.value);
    if (fileKeys.length) {
      await removeS3TTL({
        keys: fileKeys,
        bucketName: 'private',
        session
      });
    }
  });
}
```

### 5.2 获取聊天消息

**文件位置**: `packages/service/core/chat/controller.ts`

```typescript
export async function getChatItems({
  appId,
  chatId,
  offset = 0,
  limit = 50,
  field = 'dataId obj value memories userGoodFeedback userBadFeedback'
}: GetChatItemsParams): Promise<{
  histories: ChatItemType[];
  total: number;
}> {
  const [items, total] = await Promise.all([
    MongoChatItem.find({ appId, chatId })
      .sort({ createTime: 1 })
      .skip(offset)
      .limit(limit)
      .select(field)
      .lean(),
    MongoChatItem.countDocuments({ appId, chatId })
  ]);

  return {
    histories: items as ChatItemType[],
    total
  };
}

/**
 * 获取历史记录预览（用于发送给 LLM）
 */
export function getHistoryPreview(
  messages: ChatItemType[],
  previewSize = 100
): Array<{ obj: ChatRoleEnum; value: string }> {
  const result: Array<{ obj: ChatRoleEnum; value: string }> = [];

  for (let i = 0; i < messages.length; i++) {
    const msg = messages[i];
    let textContent = '';

    // 提取文本内容
    for (const item of msg.value) {
      if (item.type === 'text') {
        textContent += item.text.content;
      }
    }

    // System 消息保留完整
    if (msg.obj === ChatRoleEnum.System) {
      result.push({ obj: msg.obj, value: textContent });
      continue;
    }

    // 最后两条消息保留完整
    if (i >= messages.length - 2) {
      result.push({ obj: msg.obj, value: textContent });
      continue;
    }

    // 其他消息只保留首尾
    if (textContent.length > previewSize * 2) {
      const preview =
        textContent.slice(0, previewSize) +
        '...' +
        textContent.slice(-previewSize);
      result.push({ obj: msg.obj, value: preview });
    } else {
      result.push({ obj: msg.obj, value: textContent });
    }
  }

  return result;
}
```

```typescript
/**
 * GPT 消息格式转 FastGPT 内部格式
 */
export function GPTMessages2Chats(
  messages: ChatCompletionMessageParam[]
): ChatItemType[] {
  return messages.map((msg) => {
    // System 消息
    if (msg.role === 'system') {
      return {
        obj: ChatRoleEnum.System,
        value: [{
          type: ChatItemValueTypeEnum.text,
          text: { content: msg.content as string }
        }]
      };
    }

    // User 消息
    if (msg.role === 'user') {
      const value: UserChatItemValueItemType[] = [];

      if (typeof msg.content === 'string') {
        value.push({
          type: ChatItemValueTypeEnum.text,
          text: { content: msg.content }
        });
      } else if (Array.isArray(msg.content)) {
        for (const part of msg.content) {
          if (part.type === 'text') {
            value.push({
              type: ChatItemValueTypeEnum.text,
              text: { content: part.text }
            });
          } else if (part.type === 'image_url') {
            value.push({
              type: ChatItemValueTypeEnum.file,
              file: {
                type: 'image',
                name: 'image',
                url: part.image_url.url
              }
            });
          }
        }
      }

      return {
        obj: ChatRoleEnum.Human,
        value
      };
    }

    // Assistant 消息
    if (msg.role === 'assistant') {
      const value: AIChatItemValueItemType[] = [];

      if (msg.content) {
        value.push({
          type: ChatItemValueTypeEnum.text,
          text: { content: msg.content as string }
        });
      }

      if (msg.tool_calls) {
        for (const tool of msg.tool_calls) {
          value.push({
            type: ChatItemValueTypeEnum.tool,
            tool: {
              id: tool.id,
              name: tool.function.name,
              params: JSON.parse(tool.function.arguments),
              response: null
            }
          });
        }
      }

      return {
        obj: ChatRoleEnum.AI,
        value
      };
    }

    throw new Error(`Unknown role: ${msg.role}`);
  });
}

/**
 * FastGPT 内部格式转 GPT 消息格式
 */
export function Chats2GPTMessages(
  chats: ChatItemType[]
): ChatCompletionMessageParam[] {
  return chats.map((chat) => {
    if (chat.obj === ChatRoleEnum.System) {
      return {
        role: 'system' as const,
        content: chat.value[0]?.text?.content || ''
      };
    }

    if (chat.obj === ChatRoleEnum.Human) {
      const content: Array<ChatCompletionContentPart> = [];

      for (const item of chat.value) {
        if (item.type === 'text') {
          content.push({ type: 'text', text: item.text.content });
        } else if (item.type === 'file' && item.file.type === 'image') {
          content.push({
            type: 'image_url',
            image_url: { url: item.file.url }
          });
        }
      }

      return {
        role: 'user' as const,
        content
      };
    }

    if (chat.obj === ChatRoleEnum.AI) {
      let textContent = '';
      const toolCalls: ChatCompletionMessageToolCall[] = [];

      for (const item of chat.value) {
        if (item.type === 'text') {
          textContent += item.text.content;
        } else if (item.type === 'tool') {
          toolCalls.push({
            id: item.tool.id,
            type: 'function',
            function: {
              name: item.tool.name,
              arguments: JSON.stringify(item.tool.params)
            }
          });
        }
      }

      return {
        role: 'assistant' as const,
        content: textContent || null,
        tool_calls: toolCalls.length > 0 ? toolCalls : undefined
      };
    }

    throw new Error(`Unknown obj: ${chat.obj}`);
  });
}
```

```typescript
export function responseWrite({
  res,
  event,
  data
}: {
  res: NextApiResponse;
  event?: string;
  data: string;
}) {
  if (event) {
    res.write(`event: ${event}\n`);
  }
  res.write(`data: ${data}\n\n`);
}

/**
 * SSE 错误响应
 */
export function sseErrRes(res: NextApiResponse, error: any) {
  const processedError = processError({ error });

  responseWrite({
    res,
    event: SseResponseEventEnum.error,
    data: JSON.stringify({
      code: processedError.code,
      message: processedError.message
    })
  });
}
```

```typescript
export const getWorkflowResponseWrite = ({
  res,
  detail,
  streamResponse,
  showNodeStatus
}: WorkflowResponseWriteConfig) => {
  return ({ event, data }: { event: SseResponseEventEnum; data: any }) => {
    if (!res || res.closed || !streamResponse) return;

    // 非详情模式只发送答案
    if (!detail && !['answer', 'fastAnswer'].includes(event)) return;

    // 检查是否发送节点状态
    if (!showNodeStatus && ['flowNodeStatus', 'toolCall', 'toolParams', 'toolResponse'].includes(event)) {
      return;
    }

    responseWrite({
      res,
      event: detail ? event : undefined,
      data: JSON.stringify(data)
    });
  };
};
```
