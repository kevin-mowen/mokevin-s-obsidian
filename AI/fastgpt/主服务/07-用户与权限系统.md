# FastGPT 用户与权限系统

## 1. 概述

FastGPT 采用多租户架构设计，实现了用户(User) - 团队(Team) - 成员(TeamMember) - 组织(Org) - 成员组(MemberGroup) 的层次化用户管理体系，并通过位域(Bit-field)权限控制实现细粒度的资源访问控制。

## 2. 架构设计

### 2.1 用户体系层次

```
用户系统
├── User (用户)
│   └── 一个用户可以属于多个团队
├── Team (团队)
│   ├── Owner (拥有者)
│   └── TeamMember (团队成员)
│       ├── 状态：active/leave/forbidden
│       └── 角色：owner/admin/visitor
├── Organization (组织)
│   └── 树形结构的部门管理
└── MemberGroup (成员组)
    └── 用于权限分组管理
```

### 2.2 权限体系设计

```
权限系统
├── 资源权限 (ResourcePermission)
│   ├── 团队级别 (team)
│   ├── 应用级别 (app)
│   ├── 知识库级别 (dataset)
│   └── 模型级别 (model)
├── 角色权限 (Role)
│   ├── Owner (拥有者)
│   ├── Manager (管理者)
│   ├── Writer (编辑者)
│   └── Reader (读取者)
└── 协作者 (Collaborator)
    ├── 个人 (tmbId)
    ├── 成员组 (groupId)
    └── 组织 (orgId)
```

## 6. 权限控制器类

### 6.1 Permission 类

```typescript
// packages/global/support/permission/controller.ts
export class Permission {
  role: PermissionValueType;
  private permission: PermissionValueType = NullRoleVal;

  isOwner: boolean = false;
  hasManagePer: boolean = false;
  hasWritePer: boolean = false;
  hasReadPer: boolean = false;
  hasManageRole: boolean = false;
  hasWriteRole: boolean = false;
  hasReadRole: boolean = false;

  readonly roleList: RoleListType;
  readonly perList: PermissionListType;
  readonly rolePerMap: RolePerMapType;

  constructor({
    role = NullRoleVal,
    isOwner = false,
    roleList = CommonRoleList,
    perList = CommonPerList,
    rolePerMap = CommonRolePerMap
  }: PerConstructPros = {}) {
    if (isOwner) {
      this.role = OwnerRoleVal;
    } else {
      this.role = role;
    }

    this.roleList = roleList;
    this.perList = perList;
    this.rolePerMap = rolePerMap;
    this.updatePermissions();
  }

  // 添加角色
  addRole(...roleList: RoleValueType[]) {
    if (this.isOwner) return this;
    for (const per of roleList) {
      this.role = this.role | per;  // 位或操作
    }
    this.updatePermissions();
    return this;
  }

  // 移除角色
  removeRole(...roleList: RoleValueType[]) {
    if (this.isOwner) return this.role;
    for (const per of roleList) {
      this.role = this.role & ~per;  // 位与非操作
    }
    this.updatePermissions();
    return this;
  }

  // 检查权限
  checkPer(perm: PermissionValueType): boolean {
    if (perm === OwnerPermissionVal) {
      return this.permission === OwnerPermissionVal;
    }
    return (this.permission & perm) === perm;  // 位与检查
  }

  // 检查角色
  checkRole(role: RoleValueType): boolean {
    if (role === OwnerRoleVal) {
      return this.role === OwnerRoleVal;
    }
    return (this.role & role) === role;
  }

  // 计算权限值
  private calculatePer() {
    if (this.role === OwnerRoleVal) {
      this.permission = OwnerPermissionVal;
      return;
    }

    let role = this.role;
    this.permission = 0;
    while (role > 0) {
      // 位运算魔法：取最低位1
      this.permission |= this.rolePerMap.get(role & -role) ?? 0;
      role &= role - 1;  // 清除最低位1
    }
  }

  // 更新权限状态
  private updatePermissions() {
    this.calculatePer();
    this.isOwner = this.permission === OwnerRoleVal;
    this.hasManagePer = this.checkPer(this.roleList['manage'].value);
    this.hasWritePer = this.checkPer(this.roleList['write'].value);
    this.hasReadPer = this.checkPer(this.roleList['read'].value);
    this.hasManageRole = this.checkRole(this.roleList['manage'].value);
    this.hasWriteRole = this.checkRole(this.roleList['write'].value);
    this.hasReadRole = this.checkRole(this.roleList['read'].value);
  }
}
```

### 7.1 获取成员权限

```typescript
// packages/service/support/permission/controller.ts
export const getTmbPermission = async ({
  resourceType,
  teamId,
  tmbId,
  resourceId
}: {
  teamId: string;
  tmbId: string;
  resourceType: PerResourceTypeEnum;
  resourceId?: string;
}): Promise<PermissionValueType | undefined> => {
  // 1. 个人权限优先级最高
  const tmbPer = (
    await MongoResourcePermission.findOne(
      { resourceType, teamId, resourceId, tmbId },
      'permission'
    ).lean()
  )?.permission;

  if (tmbPer !== undefined) {
    return tmbPer;
  }

  // 2. 获取成员组权限和组织权限
  const [groupPers, orgPers] = await Promise.all([
    // 获取成员所属的所有组
    getGroupsByTmbId({ tmbId, teamId })
      .then((res) => res.map((item) => item._id))
      .then((groupIdList) =>
        MongoResourcePermission.find(
          { teamId, resourceType, groupId: { $in: groupIdList }, resourceId },
          'permission'
        ).lean()
      )
      .then((perList) => perList.map((item) => item.permission)),

    // 获取成员所属的所有组织（含父组织）
    getOrgIdSetWithParentByTmbId({ tmbId, teamId })
      .then((item) => Array.from(item))
      .then((orgIds) =>
        MongoResourcePermission.find(
          { teamId, resourceType, orgId: { $in: orgIds }, resourceId },
          'permission'
        ).lean()
      )
      .then((perList) => perList.map((item) => item.permission))
  ]);

  // 3. 合并权限（位或操作）
  return sumPer(...groupPers, ...orgPers);
};
```

### 7.2 创建默认协作者

```typescript
// packages/service/support/permission/controller.ts
export const createResourceDefaultCollaborators = async ({
  resource,
  resourceType,
  session,
  tmbId
}: {
  resource: SyncChildrenPermissionResourceType;
  resourceType: PerResourceTypeEnum;
  session: ClientSession;
  tmbId: string;
}) => {
  // 获取父资源的协作者
  const parentClbs = await getResourceOwnedClbs({
    resourceId: resource.parentId,
    resourceType,
    teamId: resource.teamId,
    session
  });

  // 构建新资源的协作者列表
  const collaborators: CollaboratorItemType[] = [
    // 继承父资源的协作者（降级Owner为Manager）
    ...parentClbs
      .filter((item) => item.tmbId !== tmbId)
      .map((clb) => {
        if (clb.permission === OwnerRoleVal) {
          clb.permission = ManageRoleVal;
        }
        return clb;
      }),
    // 创建者获得Owner权限
    {
      tmbId,
      permission: OwnerRoleVal
    }
  ];

  // 批量写入权限
  const ops = collaborators.map((clb) => ({
    updateOne: {
      filter: {
        ...pickCollaboratorIdFields(clb),
        teamId: resource.teamId,
        resourceId: resource._id,
        resourceType
      },
      update: { $set: { permission: clb.permission } },
      upsert: true
    }
  }));

  await MongoResourcePermission.bulkWrite(ops, { session });
};
```

```typescript
// packages/service/support/permission/user/auth.ts
export const authUserByToken = async (token: string): Promise<{
  userId: string;
  teamId: string;
  tmbId: string;
  isRoot: boolean;
}> => {
  // 1. 验证 JWT Token
  const payload = verifyToken(token);

  // 2. 获取用户信息
  const user = await MongoUser.findById(payload.userId);
  if (!user || user.status !== UserStatusEnum.active) {
    throw new Error('User not found or inactive');
  }

  // 3. 获取团队成员信息
  const tmb = await MongoTeamMember.findById(payload.tmbId);
  if (!tmb || tmb.status !== TeamMemberStatusEnum.active) {
    throw new Error('Team member not found or inactive');
  }

  // 4. 检查是否为Root用户
  const isRoot = await checkIsTeamOwner(tmb.teamId, tmb.userId);

  return {
    userId: String(user._id),
    teamId: String(tmb.teamId),
    tmbId: String(tmb._id),
    isRoot
  };
};
```

### 8.3 应用权限认证

```typescript
// packages/service/support/permission/app/auth.ts
export const authApp = async ({
  appId,
  tmbId,
  per
}: {
  appId: string;
  tmbId: string;
  per: PermissionValueType;
}): Promise<{
  app: AppSchema;
  permission: Permission;
}> => {
  // 1. 获取应用
  const app = await MongoApp.findById(appId).lean();
  if (!app) {
    throw new Error('App not found');
  }

  // 2. 获取成员对应用的权限
  const tmb = await MongoTeamMember.findById(tmbId);
  const isOwner = String(app.tmbId) === String(tmbId);

  const permissionValue = isOwner
    ? OwnerRoleVal
    : await getTmbPermission({
        resourceType: PerResourceTypeEnum.app,
        teamId: String(app.teamId),
        tmbId: String(tmbId),
        resourceId: String(appId)
      });

  // 3. 构建权限对象
  const permission = new Permission({
    role: permissionValue ?? NullRoleVal,
    isOwner
  });

  // 4. 验证权限
  if (!permission.checkPer(per)) {
    throw new Error('Permission denied');
  }

  return { app, permission };
};
```

## 11. 权限继承机制

### 11.1 继承权限

```typescript
// packages/service/support/permission/inheritPermission.ts
export const syncChildrenPermission = async ({
  resource,
  resourceType,
  session
}: {
  resource: SyncChildrenPermissionResourceType;
  resourceType: PerResourceTypeEnum;
  session: ClientSession;
}) => {
  // 获取所有子资源
  const children = await getChildResources(resource._id, resourceType);

  for (const child of children) {
    // 如果子资源开启了继承
    if (child.inheritPermission) {
      // 删除原有权限
      await MongoResourcePermission.deleteMany(
        { resourceId: child._id, resourceType },
        { session }
      );

      // 复制父资源权限
      const parentPermissions = await MongoResourcePermission.find(
        { resourceId: resource._id, resourceType }
      ).lean();

      await MongoResourcePermission.insertMany(
        parentPermissions.map((p) => ({
          ...p,
          _id: undefined,
          resourceId: child._id
        })),
        { session }
      );

      // 递归处理子资源的子资源
      await syncChildrenPermission({
        resource: child,
        resourceType,
        session
      });
    }
  }
};
```

## 14. 位域权限详解

### 14.1 权限值计算示例

```typescript
// 权限位定义
// manage: 0b001 = 1
// write:  0b010 = 2
// read:   0b100 = 4

// 场景1: 用户拥有读取权限
const readRole = 0b100;  // 4
// 通过 rolePerMap 映射: 0b100 -> 0b100
// 权限值: 0b100 (可读)

// 场景2: 用户拥有写入权限
const writeRole = 0b010;  // 2
// 通过 rolePerMap 映射: 0b010 -> 0b110
// 权限值: 0b110 (可写+可读)

// 场景3: 用户拥有管理权限
const manageRole = 0b001;  // 1
// 通过 rolePerMap 映射: 0b001 -> 0b111
// 权限值: 0b111 (可管理+可写+可读)

// 场景4: 合并多个权限
const combinedRole = 0b100 | 0b001;  // read + manage = 0b101 = 5
// 权限计算:
// - role & -role = 0b001 (最低位1)
// - permission |= rolePerMap.get(0b001) = 0b111
// - role &= role - 1 = 0b100
// - role & -role = 0b100
// - permission |= rolePerMap.get(0b100) = 0b100
// 最终权限值: 0b111 (manage权限包含了read)
```

### 14.2 权限检查示例

```typescript
// 检查是否有读取权限

// 检查是否有写入权限

// 检查是否有管理权限
```
