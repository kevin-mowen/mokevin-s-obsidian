```typescript
import { DashboardContainer } from '@/pageComponents/dashboard/Container';

function ContextRender() {
  return (
    <DashboardContainer>
```

**关键点**：

#### 应用类型区分逻辑

Context 根据当前路由自动决定加载哪些类型的应用：

```typescript
// 确定要加载的应用类型
const formatType = useMemo(() => {
  if (router.pathname.includes('/agent')) {
    // Agent 页面：文件夹 + 简单应用 + 工作流
    return [AppTypeEnum.folder, AppTypeEnum.simple, AppTypeEnum.workflow];
  }
  if (router.pathname.includes('/tool')) {
    // Tool 页面：工具文件夹 + 工作流工具 + MCP工具集 + HTTP工具集
    return [
      AppTypeEnum.toolFolder,
      AppTypeEnum.workflowTool,
      AppTypeEnum.mcpToolSet,
      AppTypeEnum.httpToolSet
    ];
  }
  // 其他页面（如聊天页面）显示所有类型
  return undefined;
}, [router.pathname]);
```

```typescript
// 应用列表加载
```

```typescript
const List = () => {
  const { myApps, onUpdateApp, setMoveAppId } = useContextSelector(AppListContext, (v) => v);

      }}
      gap={4}
    >
      {/* 创建按钮卡片 */}
      <ListCreateButton />

      {/* 应用卡片列表 */}
      {myApps.map((app) => (
        <AppCard
          key={app._id}
          app={app}
          onUpdate={(data) => onUpdateApp(app._id, data)}
          onMove={() => setMoveAppId(app._id)}
        />
      ))}
    </Grid>
  );
};
```

#### 应用卡片组件

```typescript
const AppCard = ({ app, onUpdate, onMove }) => {
  const router = useRouter();

  // 点击卡片的处理逻辑
  const handleClick = useCallback(() => {
    if (AppFolderTypeList.includes(app.type)) {
      // 文件夹：进入文件夹
      router.push({ query: { parentId: app._id } });
    } else if (app.permission.hasWritePer || app.permission.hasReadChatLogPer) {
      // 有编辑权限：进入编辑页面
      router.push(`/app/detail?appId=${app._id}`);
    } else {
      // 只读：打开聊天页面
      window.open(`/chat?appId=${app._id}`, '_blank');
    }
  }, [app, router]);

  return (
    <MyBox
      onClick={handleClick}
      cursor="pointer"
      _hover={{ shadow: 'md', borderColor: 'primary.300' }}
    >
      {/* 卡片头部：图标 + 名称 + 类型标签 */}
      <HStack spacing={3}>
        <Avatar src={app.avatar} size="md" />
        <Box flex={1}>
          <HStack>
            <Text fontWeight="bold" noOfLines={1}>{app.name}</Text>
            <AppTypeTag type={app.type} />
          </HStack>
        </Box>
      </HStack>

      {/* 卡片描述 */}
      <Text color="myGray.500" noOfLines={2} mt={2}>
        {app.intro || t('app:no_description')}
      </Text>

      {/* 卡片底部：创建者 + 权限 + 时间 + 操作菜单 */}
      <HStack mt={3} justify="space-between">
        <HStack spacing={2} flex={1}>
          <UserBox sourceMember={app.sourceMember} />
          <PermissionIconText private={app.private} />
          <Text fontSize="xs" color="myGray.400">
            {formatTimeToChatTime(app.updateTime)}
          </Text>
        </HStack>
```

```typescript
export const AppContextProvider = ({ children }) => {
  const router = useRouter();
  const { appId } = router.query;

  const [appDetail, setAppDetail] = useState<AppDetailType>(defaultApp);
  const [currentTab, setCurrentTab] = useState(TabEnum.appEdit);

      setAppDetail((state) => ({
        ...state,
        ...data,
        modules: data.nodes || state.modules
      }));
    }
  );

  // 发布应用版本

        ...data,
        modules: data.nodes || state.modules
      }));
```

## 7. 关键数据流

### 7.1 应用列表加载流程

```text
页面加载
    │
    ▼
AppListContextProvider 挂载
    │
    ▼
useRequest2 自动执行 getMyApps()
    │
    ├── parentId: 从 router.query 获取
    ├── type: 根据路由自动判断
    └── searchKey: 从 Context 状态获取
    │
    ▼
API: POST /core/app/list
    │
    ▼
更新 Context 状态: myApps = response
    │
    ▼
List 组件通过 useContextSelector 获取 myApps
    │
    ▼
渲染应用卡片网格
```

### 7.2 应用创建流程

```text
用户点击「创建」按钮
    │
    ▼
导航到 /dashboard/create?appType=workflow
    │
    ▼
CreateForm 渲染
    │
    ▼
用户填写表单并提交
    │
    ▼
postCreateApp() 或 postCreateMCPTools() 等
    │
    ▼
API: POST /core/app/create
    │
    ▼
返回新应用 ID
    │
    ▼
导航到 /app/detail?appId=newId
```

```text
用户在编辑器中修改

API: PUT /core/app/update
    │
    ▼
用户点击「发布」
    │
    ▼
onSaveApp({ isPublish: true })
    │
    ▼

    ▼
创建新版本记录

更新 Context 状态
```

### 9.1 团队应用模板

```typescript
// 获取团队应用模板列表（用于工作流节点选择）
export const getTeamAppTemplates = async (data?: {
  parentId?: ParentIdType;     // 父目录 ID
  searchKey?: string;          // 搜索关键词
  type?: AppTypeEnum[];        // 应用类型过滤
```

**特殊处理**：
- 如果 `parentId` 是 MCP 工具集，调用 `getMcpChildren` 获取子工具
