# FastGPT 知识库系统

> 本文档详细分析知识库模块的前端到后端完整实现

---

## 1. 功能概述

知识库系统是 FastGPT 的 RAG (Retrieval-Augmented Generation) 核心，支持：

- **多源数据导入**：文件上传、网页爬取、API 同步、飞书/语雀集成
- **智能文本分块**：段落分割、固定大小、自定义分隔符
- **向量化存储**：支持 PostgreSQL pgvector 和 Milvus
- **多模式检索**：向量搜索、全文搜索、混合搜索 + 重排序

```typescript
export async function createCollectionAndInsertData({
  collection,
  dataset,
  teamId,
  tmbId
}: CreateCollectionParams): Promise<void> {
  const { fileId, rawLink, trainingType, chunkSettings } = collection;

  // 1. 读取原始文本
  let rawText = '';
  if (fileId) {
    rawText = await readFileById(fileId);
  } else if (rawLink) {
    rawText = await readFileRawTextByUrl({ url: rawLink });
  }

  if (!rawText) {
    throw new Error('No content to process');
  }

  // 2. 文本分块
  const chunks = await rawText2Chunks({
    rawText,
    chunkSettings: chunkSettings || dataset.chunkSettings,
    collectionName: collection.name
  });

  // 3. 创建训练任务
  await pushDataListToTrainingQueue({
    teamId,
    tmbId,
    datasetId: dataset._id,
    collectionId: collection._id,
    mode: trainingType,
    chunks,
    vectorModel: dataset.vectorModel
  });
}
```

### 5.2 文本分块处理

**文件位置**: `packages/service/core/dataset/read.ts`

```typescript
/**
 * 原始文本转分块
 */
export async function rawText2Chunks({
  rawText,
  chunkSettings,
  collectionName
}: RawText2ChunksParams): Promise<ChunkItemType[]> {
  const {
    chunkSplitMode,
    chunkSize,
    chunkSplitter,
    paragraphChunkAIMode,
    paragraphChunkDeep,
    dataEnhanceCollectionName
  } = chunkSettings;

  let chunks: ChunkItemType[] = [];

  // 根据分割模式处理
  switch (chunkSplitMode) {
    case 'paragraph':
      // 段落分割
      chunks = await splitByParagraph({
        text: rawText,
        aiMode: paragraphChunkAIMode,
        deep: paragraphChunkDeep,
        maxSize: chunkSize
      });
      break;

    case 'size':
      // 固定大小分割
      chunks = splitBySize(rawText, chunkSize);
      break;

    case 'char':
      // 自定义分隔符
      chunks = splitByCustomSeparator(rawText, chunkSplitter, chunkSize);
      break;
  }

  // 数据增强：添加集合名称
  if (dataEnhanceCollectionName && collectionName) {
    chunks = chunks.map((chunk) => ({
      ...chunk,
      q: `${collectionName}\n${chunk.q}`
    }));
  }

  return chunks;
}

/**
 * 段落分割
 */
async function splitByParagraph({
  text,
  aiMode,
  deep,
  maxSize
}: {
  text: string;
  aiMode: 'auto' | 'force' | 'forbid';
  deep: number;
  maxSize: number;
}): Promise<ChunkItemType[]> {
  // 1. 按换行分割成段落
  const paragraphs = text.split(/\n\n+/);

  // 2. AI 模式处理
  if (aiMode === 'force') {
    // 强制使用 AI 识别逻辑段落
    return await aiSplitParagraphs(paragraphs, deep);
  }

  // 3. 合并小段落、分割大段落
  const chunks: ChunkItemType[] = [];
  let currentChunk = '';

  for (const para of paragraphs) {
    if ((currentChunk + para).length <= maxSize) {
      currentChunk += (currentChunk ? '\n\n' : '') + para;
    } else {
      if (currentChunk) {
        chunks.push({ q: currentChunk, a: '' });
      }
      // 如果单个段落超过 maxSize，需要进一步分割
      if (para.length > maxSize) {
        const subChunks = splitBySize(para, maxSize);
        chunks.push(...subChunks);
        currentChunk = '';
      } else {
        currentChunk = para;
      }
    }
  }

  if (currentChunk) {
    chunks.push({ q: currentChunk, a: '' });
  }

  return chunks;
}
```

```typescript
/**
 * 将数据推入训练队列
 */
export async function pushDataListToTrainingQueue({
  teamId,
  tmbId,
  datasetId,
  collectionId,
  mode,
  chunks,
  vectorModel
}: PushTrainingParams): Promise<void> {
  const model = getEmbeddingModel(vectorModel);
  const weight = model.weight || 1;

  // 批量创建训练任务
  const trainingDocs = chunks.map((chunk, index) => ({
    teamId,
    datasetId,
    collectionId,
    mode,
    q: chunk.q,
    a: chunk.a || '',
    indexes: [{ type: DatasetDataIndexTypeEnum.default, text: chunk.q }],
    weight,
    chunkIndex: index,
    lockTime: new Date(0),  // 初始未锁定
    retryCount: 0,
    expireAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000)  // 7天后过期
  }));

  await MongoDatasetTraining.insertMany(trainingDocs);
}

/**
 * 训练任务处理（由后台 Worker 调用）
 */
export async function processTrainingQueue(): Promise<void> {
  // 1. 获取待处理任务
  const task = await MongoDatasetTraining.findOneAndUpdate(
    {
      lockTime: { $lt: new Date() },
      retryCount: { $lt: 5 }
    },
    {
      $set: { lockTime: new Date(Date.now() + 5 * 60 * 1000) }  // 锁定 5 分钟
    },
    { sort: { weight: -1 } }  // 高权重优先
  );

  if (!task) return;

  try {
    // 2. 获取数据集配置
    const dataset = await MongoDataset.findById(task.datasetId).lean();
    if (!dataset) {
      await MongoDatasetTraining.deleteOne({ _id: task._id });
      return;
    }

    // 3. 生成向量
    const vectors = await getVectorsByText({
      model: getEmbeddingModel(dataset.vectorModel),
      input: task.indexes.map((idx) => idx.text)
    });

    // 4. 存储向量到向量数据库
    const vectorIds = await insertVectors({
      teamId: task.teamId,
      datasetId: task.datasetId,
      collectionId: task.collectionId,
      vectors
    });

    // 5. 创建数据记录
    await MongoDatasetData.create({
      teamId: task.teamId,
      tmbId: task.tmbId || dataset.tmbId,
      datasetId: task.datasetId,
      collectionId: task.collectionId,
      q: task.q,
      a: task.a,
      indexes: task.indexes.map((idx, i) => ({
        ...idx,
        dataId: vectorIds[i]
      })),
      chunkIndex: task.chunkIndex
    });

    // 6. 删除训练任务
    await MongoDatasetTraining.deleteOne({ _id: task._id });
  } catch (error) {
    // 7. 失败重试
    await MongoDatasetTraining.updateOne(
      { _id: task._id },
      {
        $inc: { retryCount: 1 },
        $set: {
          lockTime: new Date(0),
          errorMsg: error.message
        }
      }
    );
  }
}
```

```typescript
  // 1. 获取数据集配置
  const datasets = await MongoDataset.find({
    _id: { $in: datasetIds },
    teamId
  }).lean();

  if (datasets.length === 0) return [];

  // 2. 向量搜索
  let embeddingResults: SearchResultType[] = [];
  if (searchMode !== DatasetSearchModeEnum.fullTextRecall) {
    embeddingResults = await embeddingSearch({
      datasets,
      queries,
      limit: limit * 2  // 多取一些用于后续融合
    });
  }

  // 3. 全文搜索
  let fullTextResults: SearchResultType[] = [];
  if (searchMode !== DatasetSearchModeEnum.embedding) {
    fullTextResults = await fullTextSearch({
      teamId,
      datasetIds,
      queries,
      limit: limit * 2
    });
  }

  // 4. 结果融合
  let mergedResults: SearchResultType[];
  if (searchMode === DatasetSearchModeEnum.mixedRecall) {
    mergedResults = rrfMerge(embeddingResults, fullTextResults);
  } else if (searchMode === DatasetSearchModeEnum.embedding) {
    mergedResults = embeddingResults;
  } else {
    mergedResults = fullTextResults;
  }

  // 5. 相似度过滤
  mergedResults = mergedResults.filter((r) => r.score >= similarity);

  // 6. 重排序
  if (usingReRank && rerankModel && mergedResults.length > 0) {
    mergedResults = await reRankResults({
      query: queries[0],
      results: mergedResults,
      model: rerankModel
    });
  }

  // 7. 返回 Top-K
  return mergedResults.slice(0, limit);
}

/**
 * 向量搜索
 */
async function embeddingSearch({
  datasets,
  queries,
  limit
}: EmbeddingSearchParams): Promise<SearchResultType[]> {
  const results: SearchResultType[] = [];

  for (const dataset of datasets) {
    const model = getEmbeddingModel(dataset.vectorModel);

    // 生成查询向量
    const queryVectors = await getVectorsByText({
      model,
      input: queries,
      type: 'query'
    });

    // 向量数据库搜索
    for (const vector of queryVectors.vectors) {
      const searchResults = await vectorStore.search({
        teamId: dataset.teamId,
        datasetId: dataset._id,
        vector,
        limit
      });

      // 获取数据详情
      const dataIds = searchResults.map((r) => r.dataId);
      const dataItems = await MongoDatasetData.find({
        'indexes.dataId': { $in: dataIds }
      }).lean();

      for (const sr of searchResults) {
        const dataItem = dataItems.find((d) =>
          d.indexes.some((idx) => idx.dataId === sr.dataId)
        );
        if (dataItem) {
          results.push({
            id: dataItem._id,
            datasetId: dataset._id,
            collectionId: dataItem.collectionId,
            q: dataItem.q,
            a: dataItem.a,
            score: sr.score,
            scoreType: 'embedding'
          });
        }
      }
    }
  }

  return results;
}

/**
 * RRF 融合算法
 */
function rrfMerge(
  list1: SearchResultType[],
  list2: SearchResultType[],
  k = 60
): SearchResultType[] {
  const scoreMap = new Map<string, { item: SearchResultType; score: number }>();

  // 处理列表1
  list1.forEach((item, rank) => {
    const rrfScore = 1 / (k + rank + 1);
    scoreMap.set(item.id, { item, score: rrfScore });
  });

  // 处理列表2，合并分数
  list2.forEach((item, rank) => {
    const rrfScore = 1 / (k + rank + 1);
    const existing = scoreMap.get(item.id);
    if (existing) {
      existing.score += rrfScore;
    } else {
      scoreMap.set(item.id, { item, score: rrfScore });
    }
  });

  // 按分数排序
  return Array.from(scoreMap.values())
    .sort((a, b) => b.score - a.score)
    .map(({ item, score }) => ({ ...item, score, scoreType: 'rrf' }));
}
```

```typescript
export class PGVectorStore implements VectorStore {
  private pool: Pool;

  constructor() {
    this.pool = new Pool({
      connectionString: process.env.PG_URL
    });
  }

  async insert({
    teamId,
    datasetId,
    collectionId,
    vectors
  }: InsertParams): Promise<string[]> {
    const ids: string[] = [];

    for (const vector of vectors) {
      const id = new ObjectId().toString();
      await this.pool.query(
        `INSERT INTO dataset_vectors
         (id, team_id, dataset_id, collection_id, vector)
         VALUES ($1, $2, $3, $4, $5)`,
        [id, teamId, datasetId, collectionId, JSON.stringify(vector)]
      );
      ids.push(id);
    }

    return ids;
  }

  async search({
    teamId,
    datasetId,
    vector,
    limit
  }: SearchParams): Promise<VectorSearchResult[]> {
    const result = await this.pool.query(
      `SELECT id, 1 - (vector <=> $1) as score
       FROM dataset_vectors
       WHERE team_id = $2 AND dataset_id = $3
       ORDER BY vector <=> $1
       LIMIT $4`,
      [JSON.stringify(vector), teamId, datasetId, limit]
    );

    return result.rows.map((row) => ({
      dataId: row.id,
      score: row.score
    }));
  }

  async delete({ dataIds }: { dataIds: string[] }): Promise<void> {
    await this.pool.query(
      'DELETE FROM dataset_vectors WHERE id = ANY($1)',
      [dataIds]
    );
  }
}
```

```typescript
import { MilvusClient } from '@zilliz/milvus2-sdk-node';

export class MilvusVectorStore implements VectorStore {
  private client: MilvusClient;

  constructor() {
    this.client = new MilvusClient({
      address: process.env.MILVUS_ADDRESS
    });
  }

  async insert({
    teamId,
    datasetId,
    collectionId,
    vectors
  }: InsertParams): Promise<string[]> {
    const collectionName = `dataset_${datasetId}`;

    // 确保集合存在
    await this.ensureCollection(collectionName, vectors[0].length);

    const ids = vectors.map(() => new ObjectId().toString());

    await this.client.insert({
      collection_name: collectionName,
      data: vectors.map((vector, i) => ({
        id: ids[i],
        team_id: teamId,
        collection_id: collectionId,
        vector
      }))
    });

    return ids;
  }

  async search({
    teamId,
    datasetId,
    vector,
    limit
  }: SearchParams): Promise<VectorSearchResult[]> {
    const collectionName = `dataset_${datasetId}`;

    const results = await this.client.search({
      collection_name: collectionName,
      vector,
      limit,
      filter: `team_id == "${teamId}"`,
      output_fields: ['id']
    });

    return results.results.map((r) => ({
      dataId: r.id,
      score: r.score
    }));
  }

  private async ensureCollection(name: string, dimension: number) {
    const exists = await this.client.hasCollection({ collection_name: name });
    if (!exists) {
      await this.client.createCollection({
        collection_name: name,
        fields: [
          { name: 'id', data_type: 'VarChar', is_primary_key: true, max_length: 64 },
          { name: 'team_id', data_type: 'VarChar', max_length: 64 },
          { name: 'collection_id', data_type: 'VarChar', max_length: 64 },
          { name: 'vector', data_type: 'FloatVector', dim: dimension }
        ]
      });
    }
  }
}
```
